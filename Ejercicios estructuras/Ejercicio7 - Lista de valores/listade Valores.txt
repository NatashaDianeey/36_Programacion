// Tener en cuenta que puede haber dos o más estructuras repetitivas en un algoritmo.

}

for (i=0; i<3; i++) {

	console.log('Lista 2: ' + valoresListaDos);
	sumar = sumar + valoresListaDos;
}

if (valoresListaUno>valoresListaDos) {
